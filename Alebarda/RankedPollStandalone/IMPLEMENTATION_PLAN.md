{
    /**
     * –ü–æ–¥—Å—á–∏—Ç–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø–æ –º–µ—Ç–æ–¥—É –®—É–ª—å—Ü–µ
     *

    public function calculateWinner(array $votes, array $candidates)

    /**
     * –ü–æ—Å—Ç—Ä–æ–∏—Ç—å pairwise comparison matrix
     */
    protected function buildPairwiseMatrix(array $votes, array $candidates)
    {
        $d = [];

        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
        foreach ($candidates as $i) {
            foreach ($candidates as $j) {

            // –°—Ä–∞–≤–Ω–∏—Ç—å –∫–∞–∂–¥—É—é –ø–∞—Ä—É

                foreach ($candidates as $j) {

                    $rankJ = $ballot[$j] ?? PHP_INT_MAX;

                    // –ï—Å–ª–∏ i —Ä–∞–Ω–∂–∏—Ä–æ–≤–∞–Ω –≤—ã—à–µ j
                    if ($rankI < $rankJ) {

            }
        }

        return $d;
    }

    /**
     * –í—ã—á–∏—Å–ª–∏—Ç—å strongest paths (Floyd-Warshall)
     */
    protected function computeStrongestPaths(array $d, array $candidates)
    {
        $p = [];

        // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è

            foreach ($candidates as $j) {
                if ($i != $j) {
                    if ($d[$i][$j] > $d[$j][$i]) {
                        $p[$i][$j] = $d[$i][$j];
                    } else {
                        $p[$i][$j] = 0;
                    }
                }
            }
        }

        // Floyd-Warshall
        foreach ($candidates as $i) {
            foreach ($candidates as $j) {
                if ($i != $j) {
                    foreach ($candidates as $k) {
                        if ($i != $k && $j != $k) {
                            $p[$j][$k] = max(
                                $p[$j][$k],
                                min($p[$j][$i], $p[$i][$k])
                            );
                        }
                    }
                }
            }
        }

        return $p;
    }

    /**
     * –û–ø—Ä–µ–¥–µ–ª–∏—Ç—å —Ä–∞–Ω–∂–∏—Ä–æ–≤–∞–Ω–∏–µ
     */

        // –ü–æ–¥—Å—á–∏—Ç–∞—Ç—å —Å—á—ë—Ç –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–∞–Ω–¥–∏–¥–∞—Ç–∞

            $wins = 0;
            foreach ($candidates as $j) {
                if ($i != $j && $p[$i][$j] > $p[$j][$i]) {
                    $wins++;
                }
            }
            $scores[$i] = $wins;
        }

        // –°–æ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –ø–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤—É –ø–æ–±–µ–¥

        foreach ($scores as $candidateId => $score) {
            $ranking[] = $candidateId;
        }

        return $ranking;
    }
}
